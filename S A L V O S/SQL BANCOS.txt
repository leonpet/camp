create table public.campeonato_competicao (
  id_competicao bigint generated by default as identity not null,
  competicao text null,
  constraint campeonato_competicao_pkey primary key (id_competicao)
) TABLESPACE pg_default;

create table public.campeonato_equipe (
  id_equipe bigint generated by default as identity not null,
  nome_equipe text null,
  ranking_atual integer null,
  num_equipe integer generated by default as identity not null,
  constraint campeonato_equipe_pkey primary key (id_equipe),
  constraint campeonato_equipe_num_equipe_key unique (num_equipe)
) TABLESPACE pg_default;

create table public.campeonato_fase (
  id_fase bigint generated by default as identity not null,
  fase text not null,
  constraint campeonato_fase_pkey primary key (id_fase)
) TABLESPACE pg_default;

create table public.campeonato_gol (
  id - gol bigint generated by default as identity not null,
  jogador bigint null,
  gol_marcado integer null,
  partida bigint null,
  constraint campeonato_gol_pkey primary key ("id-gol"),
  constraint fk_jogador_gol foreign KEY (jogador) references campeonato_jogador (id_jogador) on delete CASCADE,
  constraint fk_partida_gol foreign KEY (partida) references campeonato_partida (id_partida) on delete CASCADE
) TABLESPACE pg_default;

create table public.campeonato_inscricao (
  id_inscricao bigint generated by default as identity not null,
  id_competicao bigint null,
  id_equipe bigint null,
  data_inscricao timestamp with time zone null default now(),
  grupo text null,
  id_fase bigint null,
  constraint campeonato_inscricao_pkey primary key (id_inscricao),
  constraint equipe_competicao_fase_unique unique (id_competicao, id_equipe, id_fase),
  constraint campeonato_inscricao_id_competicao_fkey foreign KEY (id_competicao) references campeonato_competicao (id_competicao) on delete CASCADE,
  constraint campeonato_inscricao_id_equipe_fkey foreign KEY (id_equipe) references campeonato_equipe (id_equipe) on delete CASCADE,
  constraint campeonato_inscricao_id_fase_fkey foreign KEY (id_fase) references campeonato_fase (id_fase)
) TABLESPACE pg_default;

create table public.campeonato_jogador (
  id_jogador bigint generated by default as identity not null,
  nome_jogador text null,
  constraint campeonato_jogador_duplicate_pkey primary key (id_jogador)
) TABLESPACE pg_default;

create table public.campeonato_partida (
  id_partida bigint generated by default as identity not null,
  id_competicao bigint null,
  grupo text null,
  id_equipe_a bigint null,
  id_equipe_b bigint null,
  gols_a integer null default 0,
  gols_b integer null default 0,
  status text null default 'Pendente'::text,
  data_partida timestamp with time zone null default now(),
  id_fase bigint null,
  constraint campeonato_partida_pkey primary key (id_partida),
  constraint campeonato_partida_id_competicao_fkey foreign KEY (id_competicao) references campeonato_competicao (id_competicao),
  constraint campeonato_partida_id_equipe_a_fkey foreign KEY (id_equipe_a) references campeonato_equipe (id_equipe),
  constraint campeonato_partida_id_equipe_b_fkey foreign KEY (id_equipe_b) references campeonato_equipe (id_equipe),
  constraint campeonato_partida_id_fase_fkey foreign KEY (id_fase) references campeonato_fase (id_fase)
) TABLESPACE pg_default;

create table public.campeonato_vinculo (
  id_vinculo bigint generated by default as identity not null,
  data_entrada timestamp with time zone not null default now(),
  numero_jogador integer null,
  numero_equipe integer null,
  constraint campeonato_vinculo_pkey primary key (id_vinculo),
  constraint fk_vinculo_equipe foreign KEY (numero_equipe) references campeonato_equipe (id_equipe),
  constraint fk_vinculo_jogador foreign KEY (numero_jogador) references campeonato_jogador (id_jogador)
) TABLESPACE pg_default;
